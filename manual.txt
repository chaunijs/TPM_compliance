1.2 Create Helper Columns
1.2.1 Create column 'Promo_Type' from conditions
    IF These text contains in column 'TPM_InvestmentDescription'
      IF "BOGO" then "BOGO"
      IF "2F1" then "2F1"
      IF "LAKSUE" then "LAKSUE"
      IF "_PO" then "PO"
1.2.2 Create column 'Promotion_Price' from conditions
    Check value in column 'Promo_Type'
    IF "LAKSUE" then  Find Number on pattern LAKSUE{Number}
    IF value in {"PO", "2F1"} then Find Number on pattern 2F1 PO{Number},PO{Number}
    Otherwise None

1.2.3 Create column 'DateRange_Days' by calculate from column ("Instore_End") - column("Instore_Start")
1.2.4 Create column 'Period'
{
  Weekly = 7,
  Bi-weekly = 14,
  Tri-weekly = 21,
  Monthly =30
}
from Number in column "DateRange_Days" with closest value among these.

1.2.5 cast column "RSP Promo" to Numeric
1.2.6 Create column "Final_Rsp"
when "RSP Promo" is not null and > 0
then use value from "RSP Promo"
otherwise use value from column "Promotion_Price"
